function compilePascal(st) {
    return st
    .replace(/\(\*[\s\S]*?\*\)/g, '')
    .replace(/PROGRAM\s+\w+/gi, '')
    .replace(/END_PROGRAM/gi, '')
    .replace(/VAR[\s\S]*?END_VAR/gi, '')
    .replace(/\bAND\b/gi, '&&')
    .replace(/\bOR\b/gi, '||')
    .replace(/\bNOT\b/gi, '!')
    .replace(/<>/g, '!=')
    .replace(/GET_VAL\(\)/gi, "parseInt(document.getElementById('iw0-slider').value)")
    .replace(/FLASH\((\d+)\)/gi, "FLASH($1)")
    .replace(/TRUE/gi, 'true')
    .replace(/FALSE/gi, 'false')
    .replace(/:=/g, '=')
    .replace(/ELSIF/gi, '} else if')
    .replace(/IF\s+(.+?)\s*THEN/gi, 'if($1){')
    .replace(/\)\s*THEN/gi, '){')
    .replace(/ELSE/gi, '}else{')
    .replace(/END_IF;?/gi, '}');
	.replace(/\bMOD\b/gi, '%')
	.replace(/\bXOR\b/gi, '^')
	.replace(/FOR\s+(\w+)\s*:=\s*(.+?)\s+TO\s+(.+?)\s+DO/gi, 'for(let $1=$2; $1<=$3; $1++){')
	.replace(/END_FOR;?/gi, '}')
	.replace(/WHILE\s+(.+?)\s+DO/gi, 'while($1){')
	.replace(/END_WHILE;?/gi, '}')
	.replace(/ABS\(/gi, 'Math.abs(')
	.replace(/SQRT\(/gi, 'Math.sqrt(')
	.replace(/SIN\(/gi, 'Math.sin(')
	.replace(/COS\(/gi, 'Math.cos(')
	.replace(/TAN\(/gi, 'Math.tan(')
	.replace(/EXP\(/gi, 'Math.exp(')
	.replace(/LN\(/gi, 'Math.log(') // În ST LN este logaritm natural, în JS este Math.log
	.replace(/LIMIT\((.+?),\s*(.+?),\s*(.+?)\)/gi, 'Math.min(Math.max($2, $1), $3)')
	.replace(/SEL\((.+?),\s*(.+?),\s*(.+?)\)/gi, '(($1) ? $3 : $2)')
	.replace(/R_TRIG\((.+?)\)/gi, 'updateR_TRIG("$1", $1)')
	.replace(/F_TRIG\((.+?)\)/gi, 'updateF_TRIG("$1", $1)')
	.replace(/TON\((.+?),\s*(.+?),\s*(.+?)\)/gi, 'updateTON("$1", $2, $3)')
	.replace(/TOF\((.+?),\s*(.+?),\s*(.+?)\)/gi, 'updateTOF("$1", $2, $3)')
	

}
